Layer Derivations in TVF

This compile-time TVF script shows how you can derive layers with TVF and simplify derivations. 
#! tvf
namespace import tvf::VERBATIM tvf::SETLAYER

VERBATIM {
  LAYOUT PATH "test.gds"
  LAYOUT PRIMARY "test"
  LAYOUT SYSTEM GDSII
  DRC RESULTS DATABASE
  drc.test.db
  DRC SUMMARY REPORT drc.summary.rpt
  LAYER OD 1
  LAYER P1 2
  LAYER P2 3
  LAYER CO 4
  LAYER M1 5
  LAYER V1 6
  LAYER M2 7
  LAYER V2 8
  LAYER M3 9
  LAYER V3 10
  LAYER M4 11
  LAYER BOUNDARY 100
}

foreach {lay} { OD PO1 PO2 CO M1 V1 M2 V2 M3 V3 M4 } {
  SETLAYER ${lay}_logo = ${lay} INSIDE BOUNDARY
  SETLAYER ${lay}_RAM = (INSIDE CELL ${lay} RAM) OUTSIDE BOUNDARY
  SETLAYER ${lay}_NRAM = (NOT INSIDE CELL ${lay} RAM) OUTSIDE BOUNDARY
  SETLAYER ${lay}_o = ${lay} OUTSIDE BOUNDARY\n
}
