General Customization Example

The following example contains many of the commands and options possible for customization files.

The control file for the run is shown after the customization file. The Customization Settings dialog box for a Calibre Interactive DRC session is shown in “Customization Overview”.

Example

#############################################################
#
#### Settings for all tools ##########
CUSTOM::SETWINGEOM 450x380

# Displays a GUI element with a text entry field for all tools.
# By default, it generates a Variable RES_TOL 1 statement in the 
# control file. 
CUSTOM::VARIABLE -name "RES_TOL" -initval 1 -prompt "Variable RES_TOL" 
# Add define for process type
CUSTOM::DEFINE -name process \
     -choices {{"wide metal" wmp} {"process 7" 7lm}} \
     -initval wmp -prompt "Add #DEFINE for process type"
############ LVS settings ##############

CUSTOM::LABEL -prompt "LVS Run Settings" -tool LVS

# Displays a checkbox in the LVS GUI. The checkbox is selected by default 
# and generates a #define MARKER_DIODE statement in the control file.
CUSTOM::DEFINE -name "MARKER_DIODE" -boolean 1 -select 1 -enable 1 \
     -prompt "Define MARKER_DIODE" -tool LVS

# Generates a Variable DEPTH ALL statement in the
# control file by default. The dropdown list includes
# ALL, PRIMARY, 1, 2, and NONE.  
# ALL, PRIMARY, and NONE  must resolve to numeric values.
CUSTOM::VARIABLE -name "DEPTH" -choices {ALL PRIMARY 1 2 NONE} \
     -prompt "Text Depth: " -tool LVS
#
###### DRC settings ########
CUSTOM::LABEL -prompt "DRC GUI Settings" -tool DRC 

# Controls for GUI runset settings, initialized to current setting.
# Need protection for runset var to prevent errors in LVS run
if {[CUSTOM::runsetVarExists drcCellName]} {

     set cellNameEntry [CUSTOM::DEFINE -name "cellName_var" \
	-select [CUSTOM::getRunsetVarValue drcCellName]  \
	-prompt "Output in cell space (GUI setting)" -tool DRC]
}
if {[CUSTOM::runsetVarExists drcStartRVE]} {		 
     set startRVEEntry [CUSTOM::DEFINE -name "startRVE_var" \
	-select [CUSTOM::getRunsetVarValue drcStartRVE]  \
	-prompt "Start RVE (GUI setting)" -tool DRC]
}
# Generates a Variable PAD_SIZE 10 statement in the control file.
#  No control is displayed in the Customization Settings dialog box.
CUSTOM::VARIABLE -name "PAD_SIZE" -initval 10 -display 0 -tool DRC
CUSTOM::SEPARATOR -tool DRC
# Defines a master control for the CUSTOM::DEFINE controls
set _masterDefines [CUSTOM::DEFINE -name "change_defines"  -select 1 \
     -prompt "Change pre-processor defines" -display 1 -tool DRC ]
CUSTOM::LABEL -prompt "Pre-Processor Defines" -tool DRC -master $_masterDefines

CUSTOM::DEFINE -name "DRC_DENSITY" -boolean 1 -select 1  \
     -prompt "Define DRC_DENSITY" -tool DRC -master $_masterDefines
CUSTOM::DEFINE -name "DRC_NAR" -boolean 1 -select 0 \
     -prompt "Define DRC_NAR" -tool DRC -master $_masterDefines 
##
## callback proc for OK button
proc myOKCallback { } {
    global cellNameEntry
    global startRVEEntry
   
    if {[CUSTOM::runsetVarExists drcCellName]} {
        set wb_cellName [$cellNameEntry cget -bvar]
   	CUSTOM::setRunsetVarValue drcCellName [set $wb_cellName]
    } 
    
    if {[CUSTOM::runsetVarExists drcStartRVE]} {
        set wb_startRVE [$startRVEEntry cget -bvar]
        CUSTOM::setRunsetVarValue drcStartRVE [set $wb_startRVE] 
    }
}
    
CUSTOM::setOKCallback myOKCallback
