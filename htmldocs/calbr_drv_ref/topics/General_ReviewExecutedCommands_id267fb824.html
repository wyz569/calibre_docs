% proc proc_cell_tree { L F {C ""} {Indent ""} } { 
	if {$C == ""} { 
		set C [$L topcell] 
	} 
	puts $F "$Indent --> $C" 
	append Indent " " 
	foreach child [$L children $C] { 
		proc_cell_tree $L $F $child $Indent 
	} 
} 
 
% proc_cell_tree $mylayout $fileID 
 
% close $fileID
