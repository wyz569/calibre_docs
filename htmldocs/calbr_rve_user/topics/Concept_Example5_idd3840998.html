Example 5: Worst Rules

This example configuration file reports worst rules. 
; Worst Rules - ChipSummary
; ByCell Histogram for Worst rules output from ChipSummary
; ByWindowColormap for Worst rules output from ChipSummary

[Common]
; Common Parameters
BinsToInclude = *
MetricsToInclude = *
OutputFormat =  html, csv, chart
FilterBin = ""
FilterMetric = Metric
FilterOperation = >=
FilterValue = 0
Rows = *
Sort = Descending
GroupBy = Metrics
; For ChipSummary Reports
GroupsToInclude = *
; For CellSummary, WindowSummary
RulesToInclude = *
; For CellSummary, CellDrillDown
IncludeFlatMetrics = 1
;For WindowSummary
IncludeWindowCoordinates = 1

[WorstRules]
ReportType = ChipSummary
GroupsToInclude = All
OutputFormat = html
[ByCellHistogramForWorstRules]
ReportType = ByCellHistogram
RulesToInclude = [WorstRules]

[ByWindowColormapForWorstRules]
ReportType = ByWindowColormap
RulesToInclude = [WorstRules]
