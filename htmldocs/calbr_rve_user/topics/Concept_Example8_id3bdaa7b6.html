Example 8: DFM Error Reports

The following example shows some common configuration setup file that generates DFM error reports and generated reports with comments. 
[Common]BinsToInclude = *
CellsToInclude = *
GroupsToInclude = All
MetricsToInclude = *
RulesToInclude = *
LinkReports = 0
[UnwaivedErrorsTable]
ReportType = ByErrorResults
BinsToInclude = 1_IMPACT
MetricsToInclude = Metric
RulesToInclude = PO.EN.1_dfm.a
GroupBy = Check/Cell
WaivedStateFilter = NotWaived
[WaivedErrors]
ReportType = ByErrorResultsSummary
BinsToInclude = 1_IMPACT
MetricsToInclude = Metric
RulesToInclude = PO.EN.1_dfm.a
GroupBy = Check/Cell
WaivedStateFilter = Waived
[WaivedErrorsColormap]
ReportType = ByErrorColormap
BinsToInclude = 1_IMPACT
MetricsToInclude = Metric
RulesToInclude = PO.EN.1_dfm.a
GroupBy = [WaivedErrors]
[WaivedErrorsHighlight]
ReportType = ByErrorHighlight
BinsToInclude = 1_IMPACT
MetricsToInclude = Metric
RulesToInclude = PO.EN.1_dfm.a
GroupBy = [WaivedErrors]

[WaivedErrorsTable]
ReportType = ByErrorResults
BinsToInclude = 1_IMPACT
MetricsToInclude = Metric
RulesToInclude = PO.EN.1_dfm.a
GroupBy = [WaivedErrors]
